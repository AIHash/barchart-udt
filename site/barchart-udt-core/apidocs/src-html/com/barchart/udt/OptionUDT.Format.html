154            /**
155             * render options in human format
156             */
157            public static enum Format {
158    
159                    DECIMAL() {
160                            @Override
161                            public String convert(final Object value) {
162                                    if (value instanceof Number) {
163                                            final long number = ((Number) value).longValue();
164                                            return String.format("%,d", number);
165                                    }
166                                    return "invalid value";
167                            }
168                    }, //
169    
170                    BINARY() {
171                            @Override
172                            public String convert(final Object value) {
173                                    if (value instanceof Number) {
174                                            final long number = ((Number) value).longValue();
175                                            return String.format("%,d (%,d K)", number, number / 1024);
176                                    }
177                                    return "invalid value";
178                            }
179                    }, //
180    
181                    BOOLEAN() {
182                            @Override
183                            public String convert(final Object value) {
184                                    if (value instanceof Boolean) {
185                                            final boolean bool = ((Boolean) value).booleanValue();
186                                            return String.format("%b", bool);
187                                    }
188                                    return "invalid value";
189                            }
190                    }, //
191    
192                    DEFAULT() {
193                            @Override
194                            public String convert(final Object value) {
195                                    return "" + value;
196                            }
197                    }, //
198    
199                    ;
200    
201                    public abstract String convert(Object value);
202    
203            }

245            /**
246             * show all option values.
247             * 
248             * @param socketUDT
249             *            the socket
250             * @param text
251             *            the text
252             */
253            public static void appendSnapshot(final SocketUDT socketUDT,
254                            final StringBuilder text) {
255    
256                    text.append("\n\t");
257                    text.append("socketID");
258                    text.append(" = ");
259                    text.append(socketUDT.socketID);
260    
261                    for (final OptionUDT option : values()) {
262                            int optionCode = 0;
263                            String optionName = null;
264                            String optionValue = null;
265                            try {
266    
267                                    optionCode = option.code;
268                                    optionName = option.name();
269                                    optionValue = option.format.convert(//
270                                                    socketUDT.getOption(option));
271    
272                                    // TODO fix this hack
273                                    if (optionName.startsWith("UD")) {
274                                            continue;
275                                    }
276    
277                                    text.append("\n\t");
278                                    text.append(optionCode);
279                                    text.append(") ");
280                                    text.append(optionName);
281                                    text.append(" = ");
282                                    text.append(optionValue);
283    
284                            } catch (final Exception e) {
285                                    log.error("unexpected; " + optionName, e);
286                            }
287    
288                    }
289    
290            }
