040package com.barchart.udt.net;
041
042import java.io.IOException;
043import java.net.InetAddress;
044import java.net.Socket;
045import java.net.SocketAddress;
046import java.net.SocketException;
047import java.net.SocketTimeoutException;
048import java.nio.channels.ServerSocketChannel;
049
050public interface IceServerSocket {
051
052        /**
053         * 
054         * Binds the <code>ServerSocket</code> to a specific address (IP address and
055         * port number).
056         * <p>
057         * If the address is <code>null</code>, then the system will pick up an
058         * ephemeral port and a valid local address to bind the socket.
059         * <p>
060         * 
061         * @param endpoint
062         *            The IP address & port number to bind to.
063         * @throws IOException
064         *             if the bind operation fails, or if the socket is already
065         *             bound.
066         * @throws SecurityException
067         *             if a <code>SecurityManager</code> is present and its
068         *             <code>checkListen</code> method doesn't allow the operation.
069         * @throws IllegalArgumentException
070         *             if endpoint is a SocketAddress subclass not supported by this
071         *             socket
072         * @since 1.4
073         */
074        void bind(SocketAddress endpoint) throws IOException;
075
076        /**
077         * 
078         * Binds the <code>ServerSocket</code> to a specific address (IP address and
079         * port number).
080         * <p>
081         * If the address is <code>null</code>, then the system will pick up an
082         * ephemeral port and a valid local address to bind the socket.
083         * <P>
084         * The <code>backlog</code> argument must be a positive value greater than
085         * 0. If the value passed if equal or less than 0, then the default value
086         * will be assumed.
087         * 
088         * @param endpoint
089         *            The IP address & port number to bind to.
090         * @param backlog
091         *            The listen backlog length.
092         * @throws IOException
093         *             if the bind operation fails, or if the socket is already
094         *             bound.
095         * @throws SecurityException
096         *             if a <code>SecurityManager</code> is present and its
097         *             <code>checkListen</code> method doesn't allow the operation.
098         * @throws IllegalArgumentException
099         *             if endpoint is a SocketAddress subclass not supported by this
100         *             socket
101         * @since 1.4
102         */
103        void bind(SocketAddress endpoint, int backlog) throws IOException;
104
105        /**
106         * Returns the local address of this server socket.
107         * 
108         * @return the address to which this socket is bound, or <code>null</code>
109         *         if the socket is unbound.
110         */
111        InetAddress getInetAddress();
112
113        /**
114         * Returns the port on which this socket is listening.
115         * 
116         * @return the port number to which this socket is listening or -1 if the
117         *         socket is not bound yet.
118         */
119        int getLocalPort();
120
121        /**
122         * Returns the address of the endpoint this socket is bound to, or
123         * <code>null</code> if it is not bound yet.
124         * 
125         * @return a <code>SocketAddress</code> representing the local endpoint of
126         *         this socket, or <code>null</code> if it is not bound yet.
127         * @see #getInetAddress()
128         * @see #getLocalPort()
129         * @see #bind(SocketAddress)
130         * @since 1.4
131         */
132
133        SocketAddress getLocalSocketAddress();
134
135        /**
136         * Listens for a connection to be made to this socket and accepts it. The
137         * method blocks until a connection is made.
138         * 
139         * <p>
140         * A new Socket <code>s</code> is created and, if there is a security
141         * manager, the security manager's <code>checkAccept</code> method is called
142         * with <code>s.getInetAddress().getHostAddress()</code> and
143         * <code>s.getPort()</code> as its arguments to ensure the operation is
144         * allowed. This could result in a SecurityException.
145         * 
146         * @exception IOException
147         *                if an I/O error occurs when waiting for a connection.
148         * @exception SecurityException
149         *                if a security manager exists and its
150         *                <code>checkAccept</code> method doesn't allow the
151         *                operation.
152         * @exception SocketTimeoutException
153         *                if a timeout was previously set with setSoTimeout and the
154         *                timeout has been reached.
155         * @exception java.nio.channels.IllegalBlockingModeException
156         *                if this socket has an associated channel, the channel is
157         *                in non-blocking mode, and there is no connection ready to
158         *                be accepted
159         * 
160         * @return the new Socket
161         * @see SecurityManager#checkAccept revised 1.4 spec JSR-51
162         */
163        Socket accept() throws IOException;
164
165        /**
166         * Closes this socket.
167         * 
168         * Any thread currently blocked in {@link #accept()} will throw a
169         * {@link SocketException}.
170         * 
171         * <p>
172         * If this socket has an associated channel then the channel is closed as
173         * well.
174         * 
175         * @exception IOException
176         *                if an I/O error occurs when closing the socket. revised
177         *                1.4 spec JSR-51
178         */
179        void close() throws IOException;
180
181        /**
182         * Returns the unique {@link java.nio.channels.ServerSocketChannel} object
183         * associated with this socket, if any.
184         * 
185         * <p>
186         * A server socket will have a channel if, and only if, the channel itself
187         * was created via the {@link java.nio.channels.ServerSocketChannel#open
188         * ServerSocketChannel.open} method.
189         * 
190         * @return the server-socket channel associated with this socket, or
191         *         <tt>null</tt> if this socket was not created for a channel
192         * 
193         * @since 1.4 spec JSR-51
194         */
195        ServerSocketChannel getChannel();
196
197        /**
198         * Returns the binding state of the ServerSocket.
199         * 
200         * @return true if the ServerSocket succesfuly bound to an address
201         * @since 1.4
202         */
203        boolean isBound();
204
205        /**
206         * Returns the closed state of the ServerSocket.
207         * 
208         * @return true if the socket has been closed
209         * @since 1.4
210         */
211        boolean isClosed();
212
213        /**
214         * Enable/disable SO_TIMEOUT with the specified timeout, in milliseconds.
215         * With this option set to a non-zero timeout, a call to accept() for this
216         * ServerSocket will block for only this amount of time. If the timeout
217         * expires, a <B>java.net.SocketTimeoutException</B> is raised, though the
218         * ServerSocket is still valid. The option <B>must</B> be enabled prior to
219         * entering the blocking operation to have effect. The timeout must be > 0.
220         * A timeout of zero is interpreted as an infinite timeout.
221         * 
222         * @param timeout
223         *            the specified timeout, in milliseconds
224         * @exception SocketException
225         *                if there is an error in the underlying protocol, such as a
226         *                TCP error.
227         * @since JDK1.1
228         * @see #getSoTimeout()
229         */
230        void setSoTimeout(int timeout) throws SocketException;
231
232        /**
233         * Retrieve setting for SO_TIMEOUT. 0 returns implies that the option is
234         * disabled (i.e., timeout of infinity).
235         * 
236         * @return the SO_TIMEOUT value
237         * @exception IOException
238         *                if an I/O error occurs
239         * @since JDK1.1
240         * @see #setSoTimeout(int)
241         */
242        int getSoTimeout() throws IOException;
243
244        /**
245         * Enable/disable the SO_REUSEADDR socket option.
246         * <p>
247         * When a TCP connection is closed the connection may remain in a timeout
248         * state for a period of time after the connection is closed (typically
249         * known as the <tt>TIME_WAIT</tt> state or <tt>2MSL</tt> wait state). For
250         * applications using a well known socket address or port it may not be
251         * possible to bind a socket to the required <tt>SocketAddress</tt> if there
252         * is a connection in the timeout state involving the socket address or
253         * port.
254         * <p>
255         * Enabling <tt>SO_REUSEADDR</tt> prior to binding the socket using
256         * {@link #bind(SocketAddress)} allows the socket to be bound even though a
257         * previous connection is in a timeout state.
258         * <p>
259         * When a <tt>ServerSocket</tt> is created the initial setting of
260         * <tt>SO_REUSEADDR</tt> is not defined. Applications can use
261         * {@link #getReuseAddress()} to determine the initial setting of
262         * <tt>SO_REUSEADDR</tt>.
263         * <p>
264         * The behaviour when <tt>SO_REUSEADDR</tt> is enabled or disabled after a
265         * socket is bound (See {@link #isBound()}) is not defined.
266         * 
267         * @param on
268         *            whether to enable or disable the socket option
269         * @exception SocketException
270         *                if an error occurs enabling or disabling the
271         *                <tt>SO_RESUEADDR</tt> socket option, or the socket is
272         *                closed.
273         * @since 1.4
274         * @see #getReuseAddress()
275         * @see #bind(SocketAddress)
276         * @see #isBound()
277         * @see #isClosed()
278         */
279        void setReuseAddress(boolean on) throws SocketException;
280
281        /**
282         * Tests if SO_REUSEADDR is enabled.
283         * 
284         * @return a <code>boolean</code> indicating whether or not SO_REUSEADDR is
285         *         enabled.
286         * @exception SocketException
287         *                if there is an error in the underlying protocol, such as a
288         *                TCP error.
289         * @since 1.4
290         * @see #setReuseAddress(boolean)
291         */
292        boolean getReuseAddress() throws SocketException;
293
294        /**
295         * Returns the implementation address and implementation port of this socket
296         * as a <code>String</code>.
297         * 
298         * @return a string representation of this socket.
299         */
300        String toString();
301
302        /**
303         * Sets a default proposed value for the SO_RCVBUF option for sockets
304         * accepted from this <tt>ServerSocket</tt>. The value actually set in the
305         * accepted socket must be determined by calling
306         * {@link Socket#getReceiveBufferSize()} after the socket is returned by
307         * {@link #accept()}.
308         * <p>
309         * The value of SO_RCVBUF is used both to set the size of the internal
310         * socket receive buffer, and to set the size of the TCP receive window that
311         * is advertized to the remote peer.
312         * <p>
313         * It is possible to change the value subsequently, by calling
314         * {@link Socket#setReceiveBufferSize(int)}. However, if the application
315         * wishes to allow a receive window larger than 64K bytes, as defined by
316         * RFC1323 then the proposed value must be set in the ServerSocket
317         * <B>before</B> it is bound to a local address. This implies, that the
318         * ServerSocket must be created with the no-argument constructor, then
319         * setReceiveBufferSize() must be called and lastly the ServerSocket is
320         * bound to an address by calling bind().
321         * <p>
322         * Failure to do this will not cause an error, and the buffer size may be
323         * set to the requested value but the TCP receive window in sockets accepted
324         * from this ServerSocket will be no larger than 64K bytes.
325         * 
326         * @exception SocketException
327         *                if there is an error in the underlying protocol, such as a
328         *                TCP error.
329         * 
330         * @param size
331         *            the size to which to set the receive buffer size. This value
332         *            must be greater than 0.
333         * 
334         * @exception IllegalArgumentException
335         *                if the value is 0 or is negative.
336         * 
337         * @since 1.4
338         * @see #getReceiveBufferSize
339         */
340        void setReceiveBufferSize(int size) throws SocketException;
341
342        /**
343         * Gets the value of the SO_RCVBUF option for this <tt>ServerSocket</tt>,
344         * that is the proposed buffer size that will be used for Sockets accepted
345         * from this <tt>ServerSocket</tt>.
346         * 
347         * <p>
348         * Note, the value actually set in the accepted socket is determined by
349         * calling {@link Socket#getReceiveBufferSize()}.
350         * 
351         * @return the value of the SO_RCVBUF option for this <tt>Socket</tt>.
352         * @exception SocketException
353         *                if there is an error in the underlying protocol, such as a
354         *                TCP error.
355         * @see #setReceiveBufferSize(int)
356         * @since 1.4
357         */
358        int getReceiveBufferSize() throws SocketException;
359
360        /**
361         * Sets performance preferences for this ServerSocket.
362         * 
363         * <p>
364         * Sockets use the TCP/IP protocol by default. Some implementations may
365         * offer alternative protocols which have different performance
366         * characteristics than TCP/IP. This method allows the application to
367         * express its own preferences as to how these tradeoffs should be made when
368         * the implementation chooses from the available protocols.
369         * 
370         * <p>
371         * Performance preferences are described by three integers whose values
372         * indicate the relative importance of short connection time, low latency,
373         * and high bandwidth. The absolute values of the integers are irrelevant;
374         * in order to choose a protocol the values are simply compared, with larger
375         * values indicating stronger preferences. If the application prefers short
376         * connection time over both low latency and high bandwidth, for example,
377         * then it could invoke this method with the values <tt>(1, 0, 0)</tt>. If
378         * the application prefers high bandwidth above low latency, and low latency
379         * above short connection time, then it could invoke this method with the
380         * values <tt>(0, 1, 2)</tt>.
381         * 
382         * <p>
383         * Invoking this method after this socket has been bound will have no
384         * effect. This implies that in order to use this capability requires the
385         * socket to be created with the no-argument constructor.
386         * 
387         * @param connectionTime
388         *            An <tt>int</tt> expressing the relative importance of a short
389         *            connection time
390         * 
391         * @param latency
392         *            An <tt>int</tt> expressing the relative importance of low
393         *            latency
394         * 
395         * @param bandwidth
396         *            An <tt>int</tt> expressing the relative importance of high
397         *            bandwidth
398         * 
399         * @since 1.5
400         */
401        void setPerformancePreferences(int connectionTime, int latency,
402                        int bandwidth);
