040package com.barchart.udt.net;
041
042import java.io.IOException;
043import java.net.DatagramPacket;
044import java.net.InetAddress;
045import java.net.PortUnreachableException;
046import java.net.SocketAddress;
047import java.net.SocketException;
048import java.net.SocketTimeoutException;
049import java.nio.channels.DatagramChannel;
050
051public interface IceDatagramSocket {
052
053        /**
054         * Binds this DatagramSocket to a specific address & port.
055         * <p>
056         * If the address is <code>null</code>, then the system will pick up an
057         * ephemeral port and a valid local address to bind the socket.
058         *<p>
059         * 
060         * @param addr
061         *            The address & port to bind to.
062         * @throws SocketException
063         *             if any error happens during the bind, or if the socket is
064         *             already bound.
065         * @throws SecurityException
066         *             if a security manager exists and its <code>checkListen</code>
067         *             method doesn't allow the operation.
068         * @throws IllegalArgumentException
069         *             if addr is a SocketAddress subclass not supported by this
070         *             socket.
071         * @since 1.4
072         */
073        void bind(SocketAddress addr) throws SocketException;
074
075        /**
076         * Connects the socket to a remote address for this socket. When a socket is
077         * connected to a remote address, packets may only be sent to or received
078         * from that address. By default a datagram socket is not connected.
079         * 
080         * <p>
081         * If the remote destination to which the socket is connected does not
082         * exist, or is otherwise unreachable, and if an ICMP destination
083         * unreachable packet has been received for that address, then a subsequent
084         * call to send or receive may throw a PortUnreachableException. Note, there
085         * is no guarantee that the exception will be thrown.
086         * 
087         * <p>
088         * A caller's permission to send and receive datagrams to a given host and
089         * port are checked at connect time. When a socket is connected, receive and
090         * send <b>will not perform any security checks</b> on incoming and outgoing
091         * packets, other than matching the packet's and the socket's address and
092         * port. On a send operation, if the packet's address is set and the
093         * packet's address and the socket's address do not match, an
094         * IllegalArgumentException will be thrown. A socket connected to a
095         * multicast address may only be used to send packets.
096         * 
097         * @param address
098         *            the remote address for the socket
099         * 
100         * @param port
101         *            the remote port for the socket.
102         * 
103         * @exception IllegalArgumentException
104         *                if the address is null, or the port is out of range.
105         * 
106         * @exception SecurityException
107         *                if the caller is not allowed to send datagrams to and
108         *                receive datagrams from the address and port.
109         * 
110         * @see #disconnect
111         * @see #send
112         * @see #receive
113         */
114        void connect(InetAddress address, int port);
115
116        /**
117         * Connects this socket to a remote socket address (IP address + port
118         * number).
119         * <p>
120         * 
121         * @param addr
122         *            The remote address.
123         * @throws SocketException
124         *             if the connect fails
125         * @throws IllegalArgumentException
126         *             if addr is null or addr is a SocketAddress subclass not
127         *             supported by this socket
128         * @since 1.4
129         * @see #connect
130         */
131        void connect(SocketAddress addr) throws SocketException;
132
133        /**
134         * Disconnects the socket. This does nothing if the socket is not connected.
135         * 
136         * @see #connect
137         */
138        void disconnect();
139
140        /**
141         * Returns the binding state of the socket.
142         * 
143         * @return true if the socket succesfuly bound to an address
144         * @since 1.4
145         */
146        boolean isBound();
147
148        /**
149         * Returns the connection state of the socket.
150         * 
151         * @return true if the socket succesfuly connected to a server
152         * @since 1.4
153         */
154        boolean isConnected();
155
156        /**
157         * Returns the address to which this socket is connected. Returns null if
158         * the socket is not connected.
159         * 
160         * @return the address to which this socket is connected.
161         */
162        InetAddress getInetAddress();
163
164        /**
165         * Returns the port for this socket. Returns -1 if the socket is not
166         * connected.
167         * 
168         * @return the port to which this socket is connected.
169         */
170        int getPort();
171
172        /**
173         * Returns the address of the endpoint this socket is connected to, or
174         * <code>null</code> if it is unconnected.
175         * 
176         * @return a <code>SocketAddress</code> representing the remote endpoint of
177         *         this socket, or <code>null</code> if it is not connected yet.
178         * @see #getInetAddress()
179         * @see #getPort()
180         * @see #connect(SocketAddress)
181         * @since 1.4
182         */
183        SocketAddress getRemoteSocketAddress();
184
185        /**
186         * Returns the address of the endpoint this socket is bound to, or
187         * <code>null</code> if it is not bound yet.
188         * 
189         * @return a <code>SocketAddress</code> representing the local endpoint of
190         *         this socket, or <code>null</code> if it is not bound yet.
191         * @see #getLocalAddress()
192         * @see #getLocalPort()
193         * @see #bind(SocketAddress)
194         * @since 1.4
195         */
196
197        SocketAddress getLocalSocketAddress();
198
199        /**
200         * Sends a datagram packet from this socket. The <code>DatagramPacket</code>
201         * includes information indicating the data to be sent, its length, the IP
202         * address of the remote host, and the port number on the remote host.
203         * 
204         * <p>
205         * If there is a security manager, and the socket is not currently connected
206         * to a remote address, this method first performs some security checks.
207         * First, if <code>p.getAddress().isMulticastAddress()</code> is true, this
208         * method calls the security manager's <code>checkMulticast</code> method
209         * with <code>p.getAddress()</code> as its argument. If the evaluation of
210         * that expression is false, this method instead calls the security
211         * manager's <code>checkConnect</code> method with arguments
212         * <code>p.getAddress().getHostAddress()</code> and <code>p.getPort()</code>
213         * . Each call to a security manager method could result in a
214         * SecurityException if the operation is not allowed.
215         * 
216         * @param p
217         *            the <code>DatagramPacket</code> to be sent.
218         * 
219         * @exception IOException
220         *                if an I/O error occurs.
221         * @exception SecurityException
222         *                if a security manager exists and its
223         *                <code>checkMulticast</code> or <code>checkConnect</code>
224         *                method doesn't allow the send.
225         * @exception PortUnreachableException
226         *                may be thrown if the socket is connected to a currently
227         *                unreachable destination. Note, there is no guarantee that
228         *                the exception will be thrown.
229         * @exception java.nio.channels.IllegalBlockingModeException
230         *                if this socket has an associated channel, and the channel
231         *                is in non-blocking mode.
232         * 
233         * @see java.net.DatagramPacket
234         * @see SecurityManager#checkMulticast(InetAddress)
235         * @see SecurityManager#checkConnect revised 1.4 spec JSR-51
236         */
237        void send(DatagramPacket p) throws IOException;
238
239        /**
240         * Receives a datagram packet from this socket. When this method returns,
241         * the <code>DatagramPacket</code>'s buffer is filled with the data
242         * received. The datagram packet also contains the sender's IP address, and
243         * the port number on the sender's machine.
244         * <p>
245         * This method blocks until a datagram is received. The <code>length</code>
246         * field of the datagram packet object contains the length of the received
247         * message. If the message is longer than the packet's length, the message
248         * is truncated.
249         * <p>
250         * If there is a security manager, a packet cannot be received if the
251         * security manager's <code>checkAccept</code> method does not allow it.
252         * 
253         * @param p
254         *            the <code>DatagramPacket</code> into which to place the
255         *            incoming data.
256         * @exception IOException
257         *                if an I/O error occurs.
258         * @exception SocketTimeoutException
259         *                if setSoTimeout was previously called and the timeout has
260         *                expired.
261         * @exception PortUnreachableException
262         *                may be thrown if the socket is connected to a currently
263         *                unreachable destination. Note, there is no guarantee that
264         *                the exception will be thrown.
265         * @exception java.nio.channels.IllegalBlockingModeException
266         *                if this socket has an associated channel, and the channel
267         *                is in non-blocking mode.
268         * @see java.net.DatagramPacket
269         * @see java.net.DatagramSocket revised 1.4 spec JSR-51
270         */
271        void receive(DatagramPacket p) throws IOException;
272
273        /**
274         * Gets the local address to which the socket is bound.
275         * 
276         * <p>
277         * If there is a security manager, its <code>checkConnect</code> method is
278         * first called with the host address and <code>-1</code> as its arguments
279         * to see if the operation is allowed.
280         * 
281         * @see SecurityManager#checkConnect
282         * @return the local address to which the socket is bound, or an
283         *         <code>InetAddress</code> representing any local address if either
284         *         the socket is not bound, or the security manager
285         *         <code>checkConnect</code> method does not allow the operation
286         * @since 1.1
287         */
288        InetAddress getLocalAddress();
289
290        /**
291         * Returns the port number on the local host to which this socket is bound.
292         * 
293         * @return the port number on the local host to which this socket is bound.
294         */
295        int getLocalPort();
296
297        /**
298         * Enable/disable SO_TIMEOUT with the specified timeout, in milliseconds.
299         * With this option set to a non-zero timeout, a call to receive() for this
300         * DatagramSocket will block for only this amount of time. If the timeout
301         * expires, a <B>java.net.SocketTimeoutException</B> is raised, though the
302         * DatagramSocket is still valid. The option <B>must</B> be enabled prior to
303         * entering the blocking operation to have effect. The timeout must be > 0.
304         * A timeout of zero is interpreted as an infinite timeout.
305         * 
306         * @param timeout
307         *            the specified timeout in milliseconds.
308         * @throws SocketException
309         *             if there is an error in the underlying protocol, such as an
310         *             UDP error.
311         * @since JDK1.1
312         * @see #getSoTimeout()
313         */
314        void setSoTimeout(int timeout) throws SocketException;
315
316        /**
317         * Retrieve setting for SO_TIMEOUT. 0 returns implies that the option is
318         * disabled (i.e., timeout of infinity).
319         * 
320         * @return the setting for SO_TIMEOUT
321         * @throws SocketException
322         *             if there is an error in the underlying protocol, such as an
323         *             UDP error.
324         * @since JDK1.1
325         * @see #setSoTimeout(int)
326         */
327        int getSoTimeout() throws SocketException;
328
329        /**
330         * Sets the SO_SNDBUF option to the specified value for this
331         * <tt>DatagramSocket</tt>. The SO_SNDBUF option is used by the network
332         * implementation as a hint to size the underlying network I/O buffers. The
333         * SO_SNDBUF setting may also be used by the network implementation to
334         * determine the maximum size of the packet that can be sent on this socket.
335         * <p>
336         * As SO_SNDBUF is a hint, applications that want to verify what size the
337         * buffer is should call {@link #getSendBufferSize()}.
338         * <p>
339         * Increasing the buffer size may allow multiple outgoing packets to be
340         * queued by the network implementation when the send rate is high.
341         * <p>
342         * Note: If {@link #send(DatagramPacket)} is used to send a
343         * <code>DatagramPacket</code> that is larger than the setting of SO_SNDBUF
344         * then it is implementation specific if the packet is sent or discarded.
345         * 
346         * @param size
347         *            the size to which to set the send buffer size. This value must
348         *            be greater than 0.
349         * 
350         * @exception SocketException
351         *                if there is an error in the underlying protocol, such as
352         *                an UDP error.
353         * @exception IllegalArgumentException
354         *                if the value is 0 or is negative.
355         * @see #getSendBufferSize()
356         */
357        void setSendBufferSize(int size) throws SocketException;
358
359        /**
360         * Get value of the SO_SNDBUF option for this <tt>DatagramSocket</tt>, that
361         * is the buffer size used by the platform for output on this
362         * <tt>DatagramSocket</tt>.
363         * 
364         * @return the value of the SO_SNDBUF option for this
365         *         <tt>DatagramSocket</tt>
366         * @exception SocketException
367         *                if there is an error in the underlying protocol, such as
368         *                an UDP error.
369         * @see #setSendBufferSize
370         */
371        int getSendBufferSize() throws SocketException;
372
373        /**
374         * Sets the SO_RCVBUF option to the specified value for this
375         * <tt>DatagramSocket</tt>. The SO_RCVBUF option is used by the the network
376         * implementation as a hint to size the underlying network I/O buffers. The
377         * SO_RCVBUF setting may also be used by the network implementation to
378         * determine the maximum size of the packet that can be received on this
379         * socket.
380         * <p>
381         * Because SO_RCVBUF is a hint, applications that want to verify what size
382         * the buffers were set to should call {@link #getReceiveBufferSize()}.
383         * <p>
384         * Increasing SO_RCVBUF may allow the network implementation to buffer
385         * multiple packets when packets arrive faster than are being received using
386         * {@link #receive(DatagramPacket)}.
387         * <p>
388         * Note: It is implementation specific if a packet larger than SO_RCVBUF can
389         * be received.
390         * 
391         * @param size
392         *            the size to which to set the receive buffer size. This value
393         *            must be greater than 0.
394         * 
395         * @exception SocketException
396         *                if there is an error in the underlying protocol, such as
397         *                an UDP error.
398         * @exception IllegalArgumentException
399         *                if the value is 0 or is negative.
400         * @see #getReceiveBufferSize()
401         */
402        void setReceiveBufferSize(int size) throws SocketException;
403
404        /**
405         * Get value of the SO_RCVBUF option for this <tt>DatagramSocket</tt>, that
406         * is the buffer size used by the platform for input on this
407         * <tt>DatagramSocket</tt>.
408         * 
409         * @return the value of the SO_RCVBUF option for this
410         *         <tt>DatagramSocket</tt>
411         * @exception SocketException
412         *                if there is an error in the underlying protocol, such as
413         *                an UDP error.
414         * @see #setReceiveBufferSize(int)
415         */
416        int getReceiveBufferSize() throws SocketException;

418        /**
419         * Enable/disable the SO_REUSEADDR socket option.
420         * <p>
421         * For UDP sockets it may be necessary to bind more than one socket to the
422         * same socket address. This is typically for the purpose of receiving
423         * multicast packets (See {@link java.net.MulticastSocket}). The
424         * <tt>SO_REUSEADDR</tt> socket option allows multiple sockets to be bound
425         * to the same socket address if the <tt>SO_REUSEADDR</tt> socket option is
426         * enabled prior to binding the socket using {@link #bind(SocketAddress)}.
427         * <p>
428         * Note: This functionality is not supported by all existing platforms, so
429         * it is implementation specific whether this option will be ignored or not.
430         * However, if it is not supported then {@link #getReuseAddress()} will
431         * always return <code>false</code>.
432         * <p>
433         * When a <tt>DatagramSocket</tt> is created the initial setting of
434         * <tt>SO_REUSEADDR</tt> is disabled.
435         * <p>
436         * The behaviour when <tt>SO_REUSEADDR</tt> is enabled or disabled after a
437         * socket is bound (See {@link #isBound()}) is not defined.
438         * 
439         * @param on
440         *            whether to enable or disable the
441         * @exception SocketException
442         *                if an error occurs enabling or disabling the
443         *                <tt>SO_RESUEADDR</tt> socket option, or the socket is
444         *                closed.
445         * @since 1.4
446         * @see #getReuseAddress()
447         * @see #bind(SocketAddress)
448         * @see #isBound()
449         * @see #isClosed()
450         */
451        void setReuseAddress(boolean on) throws SocketException;
452
453        /**
454         * Tests if SO_REUSEADDR is enabled.
455         * 
456         * @return a <code>boolean</code> indicating whether or not SO_REUSEADDR is
457         *         enabled.
458         * @exception SocketException
459         *                if there is an error in the underlying protocol, such as
460         *                an UDP error.
461         * @since 1.4
462         * @see #setReuseAddress(boolean)
463         */
464        boolean getReuseAddress() throws SocketException;
465

467         * Enable/disable SO_BROADCAST.
468         * 
469         * @param on
470         *            whether or not to have broadcast turned on.
471         * @exception SocketException
472         *                if there is an error in the underlying protocol, such as
473         *                an UDP error.
474         * @since 1.4
475         * @see #getBroadcast()
476         */
477        void setBroadcast(boolean on) throws SocketException;

480         * Tests if SO_BROADCAST is enabled.

482         * @return a <code>boolean</code> indicating whether or not SO_BROADCAST is

484         * @exception SocketException
485         *                if there is an error in the underlying protocol, such as

487         * @since 1.4
488         * @see #setBroadcast(boolean)
489         */
490        boolean getBroadcast() throws SocketException;

492        /**
493         * Sets traffic class or type-of-service octet in the IP datagram header for
494         * datagrams sent from this DatagramSocket. As the underlying network
495         * implementation may ignore this value applications should consider it a
496         * hint.
497         * 
498         * <P>
499         * The tc <B>must</B> be in the range <code> 0 <= tc <=
500         * 255</code> or an IllegalArgumentException will be thrown.
501         * <p>
502         * Notes:
503         * <p>
504         * for Internet Protocol v4 the value consists of an octet with precedence
505         * and TOS fields as detailed in RFC 1349. The TOS field is bitset created
506         * by bitwise-or'ing values such the following :-
507         * <p>
508         * <UL>
509         * <LI><CODE>IPTOS_LOWCOST (0x02)</CODE></LI>
510         * <LI><CODE>IPTOS_RELIABILITY (0x04)</CODE></LI>
511         * <LI><CODE>IPTOS_THROUGHPUT (0x08)</CODE></LI>
512         * <LI><CODE>IPTOS_LOWDELAY (0x10)</CODE></LI>
513         * </UL>
514         * The last low order bit is always ignored as this corresponds to the MBZ
515         * (must be zero) bit.
516         * <p>
517         * Setting bits in the precedence field may result in a SocketException
518         * indicating that the operation is not permitted.
519         * <p>
520         * for Internet Protocol v6 <code>tc</code> is the value that would be
521         * placed into the sin6_flowinfo field of the IP header.
522         * 
523         * @param tc
524         *            an <code>int</code> value for the bitset.
525         * @throws SocketException
526         *             if there is an error setting the traffic class or
527         *             type-of-service
528         * @since 1.4
529         * @see #getTrafficClass
530         */
531        void setTrafficClass(int tc) throws SocketException;
532
533        /**
534         * Gets traffic class or type-of-service in the IP datagram header for
535         * packets sent from this DatagramSocket.
536         * <p>
537         * As the underlying network implementation may ignore the traffic class or
538         * type-of-service set using {@link #setTrafficClass(int)} this method may
539         * return a different value than was previously set using the
540         * {@link #setTrafficClass(int)} method on this DatagramSocket.
541         * 
542         * @return the traffic class or type-of-service already set
543         * @throws SocketException
544         *             if there is an error obtaining the traffic class or
545         *             type-of-service value.
546         * @since 1.4
547         * @see #setTrafficClass(int)
548         */
549        int getTrafficClass() throws SocketException;
550
551        /**
552         * Closes this datagram socket.
553         * <p>
554         * Any thread currently blocked in {@link #receive} upon this socket will
555         * throw a {@link SocketException}.
556         * 
557         * <p>
558         * If this socket has an associated channel then the channel is closed as
559         * well.
560         * 
561         * revised 1.4 spec JSR-51
562         */
563        void close();
564
565        /**
566         * Returns whether the socket is closed or not.
567         * 
568         * @return true if the socket has been closed
569         * @since 1.4
570         */
571        boolean isClosed();
572
573        /**
574         * Returns the unique {@link java.nio.channels.DatagramChannel} object
575         * associated with this datagram socket, if any.
576         * 
577         * <p>
578         * A datagram socket will have a channel if, and only if, the channel itself
579         * was created via the {@link java.nio.channels.DatagramChannel#open
580         * DatagramChannel.open} method.
581         * 
582         * @return the datagram channel associated with this datagram socket, or
583         *         <tt>null</tt> if this socket was not created for a channel
584         * 
585         * @since 1.4 spec JSR-51
586         */
587        DatagramChannel getChannel();
