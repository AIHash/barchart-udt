040package com.barchart.udt.net;
041
042import java.io.IOException;
043import java.net.InetAddress;
044import java.net.InetSocketAddress;
045import java.net.ServerSocket;
046import java.net.Socket;
047import java.net.SocketAddress;
048import java.net.SocketException;
049import java.nio.channels.ServerSocketChannel;
050
051import com.barchart.udt.ExceptionUDT;
052import com.barchart.udt.SocketUDT;
053import com.barchart.udt.TypeUDT;
054
055public class NetServerSocketUDT extends ServerSocket implements IceServerSocket {
056
057        protected final SocketUDT serverSocketUDT;
058
059        public NetServerSocketUDT() throws IOException {
060                this.serverSocketUDT = new SocketUDT(TypeUDT.STREAM);
061                this.serverSocketUDT.configureBlocking(true);
062        }
063
064        // exception thanks to JDK designers
065        /** NOTE: you just carefully choose TypeUDT */
066        public NetServerSocketUDT(final SocketUDT socketUDT) throws IOException {
067                this.serverSocketUDT = socketUDT;
068        }
069
070        //
071
072        @Override
073        public Socket accept() throws IOException {
074                final SocketUDT connector = serverSocketUDT.accept();
075                return new NetSocketUDT(connector);
076        }
077
078        @Override
079        public void bind(SocketAddress endpoint) throws IOException {
080                final int backlog = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
081                bind(endpoint, backlog);
082        }
083
084        // NOTE: bind() means listen() for UDT server socket
085        /*
086         * The listen method lets a UDT socket enter listening state. The socket
087         * must call bind before a listen call.
088         */
089        @Override
090        public void bind(SocketAddress bindpoint, int backlog) throws IOException {
091                if (bindpoint == null) {
092                        bindpoint = new InetSocketAddress(0);
093                }
094                if (backlog <= 0) {
095                        backlog = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
096                }
097                serverSocketUDT.bind((InetSocketAddress) bindpoint);
098                serverSocketUDT.listen(backlog);
099        }
100
101        @Override
102        public void close() throws IOException {
103                serverSocketUDT.close();
104        }
105
106        @Override
107        public ServerSocketChannel getChannel() {
108                throw new RuntimeException("feature not available");
109        }
110
111        @Override
112        public InetAddress getInetAddress() {
113                return serverSocketUDT.getLocalInetAddress();
114        }
115
116        @Override
117        public int getLocalPort() {
118                return serverSocketUDT.getLocalInetPort();
119        }
120
121        @Override
122        public SocketAddress getLocalSocketAddress() {
123                try {
124                        return serverSocketUDT.getLocalSocketAddress();
125                } catch (ExceptionUDT e) {
126                        return null;
127                }
128        }
129
130        @Override
131        public int getReceiveBufferSize() throws SocketException {
132                return serverSocketUDT.getReceiveBufferSize();
133        }
134
135        @Override
136        public boolean getReuseAddress() throws SocketException {
137                return serverSocketUDT.getReuseAddress();
138        }
139
140        @Override
141        public int getSoTimeout() throws IOException {
142                return serverSocketUDT.getSoTimeout();
143        }
144
145        @Override
146        public boolean isBound() {
147                return serverSocketUDT.isBound();
148        }
149
150        @Override
151        public boolean isClosed() {
152                return serverSocketUDT.isClosed();
153        }
154
155        @Override
156        public void setPerformancePreferences(int connectionTime, int latency,
157                        int bandwidth) {
158                throw new RuntimeException("feature not available");
159        }
160
161        // NOTE: set both send and receive, since they are inherited on accept()
162        @Override
163        public void setReceiveBufferSize(int size) throws SocketException {
164                serverSocketUDT.setReceiveBufferSize(size);
165                serverSocketUDT.setSendBufferSize(size);
166        }
167
168        @Override
169        public void setReuseAddress(boolean on) throws SocketException {
170                serverSocketUDT.setReuseAddress(on);
171        }
172
173        @Override
174        public void setSoTimeout(int timeout) throws SocketException {
175                serverSocketUDT.setSoTimeout(timeout);
176        }
177
178}
