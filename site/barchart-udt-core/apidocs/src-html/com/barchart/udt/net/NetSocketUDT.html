008    package com.barchart.udt.net;
009    
010    import java.io.IOException;
011    import java.io.InputStream;
012    import java.io.OutputStream;
013    import java.net.InetAddress;
014    import java.net.InetSocketAddress;
015    import java.net.Socket;
016    import java.net.SocketAddress;
017    import java.net.SocketException;
018    import java.nio.channels.SocketChannel;
019    
020    import com.barchart.udt.ExceptionUDT;
021    import com.barchart.udt.SocketUDT;
022    import com.barchart.udt.TypeUDT;
023    import com.barchart.udt.anno.ThreadSafe;
024    
025    /**
026     * {@link Socket} - like wrapper for {@link SocketUDT}
027     */
028    public class NetSocketUDT extends Socket implements IceSocket, IceCommon {
029    
030            @ThreadSafe("this")
031            protected InputStream inputStream;
032            @ThreadSafe("this")
033            protected OutputStream outputStream;
034    
035            protected final SocketUDT socketUDT;
036    
037            /** uses {@link TypeUDT#STREAM} socket in blocking mode */
038            public NetSocketUDT() throws ExceptionUDT {
039                    this(new SocketUDT(TypeUDT.STREAM));
040                    this.socketUDT.setBlocking(true);
041            }
042    
043            /** uses provided socket keeping blocking mode */
044            protected NetSocketUDT(final SocketUDT socketUDT) throws ExceptionUDT {
045                    this.socketUDT = socketUDT;
046            }
047    
048            @Override
049            public void bind(SocketAddress bindpoint) throws IOException {
050                    if (bindpoint == null) {
051                            bindpoint = new InetSocketAddress(0);
052                    }
053                    socketUDT.bind((InetSocketAddress) bindpoint);
054            }
055    
056            @Override
057            public void close() throws IOException {
058                    socketUDT.close();
059            }
060    
061            @Override
062            public void connect(final SocketAddress endpoint) throws IOException {
063                    socketUDT.connect((InetSocketAddress) endpoint);
064            }
065    
066            @Override
067            public void connect(final SocketAddress endpoint, final int timeout)
068                            throws IOException {
069                    throw new RuntimeException("feature not available");
070            }
071    
072            @Override
073            public SocketChannel getChannel() {
074                    throw new RuntimeException("feature not available");
075            }
076    
077            @Override
078            public InetAddress getInetAddress() {
079                    return socketUDT.getRemoteInetAddress();
080            }
081    
082            @Override
083            public synchronized InputStream getInputStream() throws IOException {
084                    if (inputStream == null) {
085                            inputStream = new NetInputStreamUDT(socketUDT);
086                    }
087                    return inputStream;
088            }
089    
090            @Override
091            public boolean getKeepAlive() throws SocketException {
092                    throw new RuntimeException("feature not available");
093            }
094    
095            @Override
096            public InetAddress getLocalAddress() {
097                    return socketUDT.getLocalInetAddress();
098            }
099    
100            @Override
101            public int getLocalPort() {
102                    return socketUDT.getLocalInetPort();
103            }
104    
105            @Override
106            public SocketAddress getLocalSocketAddress() {
107                    try {
108                            return socketUDT.getLocalSocketAddress();
109                    } catch (final ExceptionUDT e) {
110                            return null;
111                    }
112            }
113    
114            @Override
115            public boolean getOOBInline() throws SocketException {
116                    throw new RuntimeException("feature not available");
117            }
118    
119            @Override
120            public synchronized OutputStream getOutputStream() throws IOException {
121                    if (outputStream == null) {
122                            outputStream = new NetOutputStreamUDT(socketUDT);
123                    }
124                    return outputStream;
125            }
126    
127            @Override
128            public int getPort() {
129                    return socketUDT.getRemoteInetPort();
130            }
131    
132            @Override
133            public int getReceiveBufferSize() throws SocketException {
134                    return socketUDT.getReceiveBufferSize();
135            }
136    
137            @Override
138            public SocketAddress getRemoteSocketAddress() {
139                    try {
140                            return socketUDT.getRemoteSocketAddress();
141                    } catch (final ExceptionUDT e) {
142                            return null;
143                    }
144            }
145    
146            @Override
147            public boolean getReuseAddress() throws SocketException {
148                    return socketUDT.getReuseAddress();
149            }
150    
151            @Override
152            public int getSendBufferSize() throws SocketException {
153                    return socketUDT.getSendBufferSize();
154            }
155    
156            @Override
157            public int getSoLinger() throws SocketException {
158                    return socketUDT.getSoLinger();
159            }
160    
161            @Override
162            public int getSoTimeout() throws SocketException {
163                    return socketUDT.getSoTimeout();
164            }
165    
166            @Override
167            public boolean getTcpNoDelay() throws SocketException {
168                    throw new RuntimeException("feature not available");
169            }
170    
171            @Override
172            public int getTrafficClass() throws SocketException {
173                    throw new RuntimeException("feature not available");
174            }
175    
176            @Override
177            public boolean isBound() {
178                    return socketUDT.isBound();
179            }
180    
181            @Override
182            public boolean isClosed() {
183                    return socketUDT.isClosed();
184            }
185    
186            @Override
187            public boolean isConnected() {
188                    return socketUDT.isConnected();
189            }
190    
191            @Override
192            public boolean isInputShutdown() {
193                    return socketUDT.isClosed();
194            }
195    
196            @Override
197            public boolean isOutputShutdown() {
198                    return socketUDT.isClosed();
199            }
200    
201            @Override
202            public void sendUrgentData(final int data) throws IOException {
203                    throw new RuntimeException("feature not available");
204            }
205    
206            @Override
207            public void setKeepAlive(final boolean on) throws SocketException {
208                    throw new RuntimeException("feature not available");
209            }
210    
211            @Override
212            public void setOOBInline(final boolean on) throws SocketException {
213                    throw new RuntimeException("feature not available");
214            }
215    
216            @Override
217            public void setPerformancePreferences(final int connectionTime,
218                            final int latency, final int bandwidth) {
219                    throw new RuntimeException("feature not available");
220            }
221    
222            @Override
223            public void setReceiveBufferSize(final int size) throws SocketException {
224                    socketUDT.setReceiveBufferSize(size);
225            }
226    
227            @Override
228            public void setReuseAddress(final boolean on) throws SocketException {
229                    socketUDT.setReuseAddress(on);
230            }
231    
232            @Override
233            public void setSendBufferSize(final int size) throws SocketException {
234                    socketUDT.setSendBufferSize(size);
235            }
236    
237            @Override
238            public void setSoLinger(final boolean on, final int linger)
239                            throws SocketException {
240                    socketUDT.setSoLinger(on, linger);
241            }
242    
243            @Override
244            public void setSoTimeout(final int timeout) throws SocketException {
245                    socketUDT.setSoTimeout(timeout);
246            }
247    
248            @Override
249            public void setTcpNoDelay(final boolean on) throws SocketException {
250                    throw new RuntimeException("feature not available");
251            }
252    
253            @Override
254            public void setTrafficClass(final int tc) throws SocketException {
255                    throw new RuntimeException("feature not available");
256            }
257    
258            @Override
259            public void shutdownInput() throws IOException {
260                    socketUDT.close();
261            }
262    
263            @Override
264            public void shutdownOutput() throws IOException {
265                    socketUDT.close();
266            }
267    
268            @Override
269            public SocketUDT socketUDT() {
270                    return socketUDT;
271            }
272    
273    }
