180        /** The log. */

183        /**
184         * show all option values.
185         * 
186         * @param socketUDT
187         *            the socket
188         * @param text
189         *            the text
190         */
191        public static void appendSnapshot(final SocketUDT socketUDT,
192                        final StringBuilder text) {
193
194                text.append("\n\t");
195                text.append("socketID");
196                text.append(" = ");
197                text.append(socketUDT.socketID);
198
199                for (OptionUDT option : values()) {
200                        int optionCode = 0;
201                        String optionName = null;
202                        String optionValue = null;
203                        try {
204
205                                optionCode = option.code;
206                                optionName = option.name();
207                                optionValue = option.format.convert(//
208                                                socketUDT.getOption(option));
209
210                                // TODO fix this hack
211                                if (optionName.startsWith("UD")) {
212                                        continue;
213                                }
214
215                                text.append("\n\t");
216                                text.append(optionCode);
217                                text.append(") ");
218                                text.append(optionName);
219                                text.append(" = ");
220                                text.append(optionValue);
221
222                        } catch (Exception e) {
223                                log.error("unexpected; " + optionName, e);
224                        }
225
226                }
227
228        }
