059/**
060 * you must use {@link SelectorProviderUDT#openSocketChannel()} to obtain
061 * instance of this class; do not use JDK
062 * {@link java.nio.channels.SocketChannel#open()};
063 * 
064 * example:
065 * 
066 * [code]
067 * 
068 * SelectorProvider provider = SelectorProviderUDT.DATAGRAM;
069 * 
070 * SocketChannel clientChannel = provider.openSocketChannel();
071 * 
072 * clientChannel.configureBlocking(true);
073 * 
074 * Socket clientSocket = clientChannel.socket();
075 * 
076 * InetSocketAddress clientAddress = new InetSocketAddress("localhost", 10000);
077 * 
078 * clientSocket.bind(clientAddress);
079 * 
080 * assert clientSocket.isBound();
081 * 
082 * InetSocketAddress serverAddress = new InetSocketAddress("localhost", 12345);
083 * 
084 * clientChannel.connect(serverAddress);
085 * 
086 * assert clientSocket.isConnected();
087 * 
088 * [/code]
089 */
090public class ChannelSocketUDT extends SocketChannel implements ChannelUDT {

107        /*
108         * local volatile variable, which mirrors super.blocking, to avoid the cost
109         * of synchronized call inside isBlocking()
110         */
111        private volatile boolean isBlockingMode = isBlocking();
112
113        @Override
114        protected void implConfigureBlocking(boolean block) throws IOException {
115                socketUDT.configureBlocking(block);
116                isBlockingMode = block;
117        }

119        private volatile boolean isConnectionPending;
120
121        @Override
122        public boolean connect(SocketAddress remote) throws IOException {
123
124                if (!isOpen()) {
125                        throw new ClosedChannelException();
126                }
127
128                if (isConnected()) {
129                        log.warn("already connected; ignoring remote={}", remote);
130                        return true;
131                }
132
133                if (remote == null) {
134                        close();
135                        throw new NullPointerException("remote == null");
136                }
137
138                InetSocketAddress remoteSocket = (InetSocketAddress) remote;
139                if (remoteSocket.isUnresolved()) {
140                        log.error("can not use unresolved address: remote={}", remote);
141                        close();
142                        throw new UnresolvedAddressException();
143                }
144
145                if (isBlocking()) {
146                        if (isConnectionPending) {
147                                close();
148                                throw new ConnectionPendingException();
149                        } else {
150                                isConnectionPending = true;
151                        }
152                        try {
153                                begin();
154                                socketUDT.connect(remoteSocket);
155                        } finally {
156                                end(true);
157                        }
158                        isConnectionPending = false;
159                        return socketUDT.isConnected();
160                } else { // non Blocking
161                        final SelectionKeyUDT key = channelKey;
162                        if (key == null) {
163                                // this channel is independent of any selector
164                                log.error("UDT channel is in non blocking mode;"
165                                                + "must register with a selector "
166                                                + "before trying to connect()");
167                                throw new IllegalBlockingModeException();
168                        } else {
169                                // this channel is registered with a selector
170                                key.selectorUDT.submitConnectRequest(key, remoteSocket);
171                                /*
172                                 * connection operation must later be completed by invoking the
173                                 * finishConnect() method.
174                                 */
175                                return false;
176                        }
177                }
178
179        }
180
181        /**
182         * note this is redundant for blocking mode
183         */
184        @Override
185        public boolean finishConnect() throws IOException {
186                if (!isOpen()) {
187                        throw new ClosedChannelException();
188                }
189                if (isBlocking()) {
190                        return isConnected();
191                } else { // non blocking
192                        if (isConnected()) {
193                                logStatus();
194                                return true;
195                        } else {
196                                IOException exception = connectException;
197                                if (exception == null) {
198                                        return false;
199                                } else {
200                                        throw exception;
201                                }
202                        }
203                }
204        }

216        /**
217         * See {@link java.nio.channels.SocketChannel#read(ByteBuffer)} contract;
218         * note: this method does not return (-1) as EOS (end of stream flag)
219         * 
220         * @return <code><0</code> should not happen<br>
221         *         <code>=0</code> blocking mode: timeout occurred on receive<br>
222         *         <code>=0</code> non-blocking mode: nothing is received by the
223         *         underlying UDT socket<br>
224         *         <code>>0</code> actual bytes received count<br>
225         * @see com.barchart.udt.SocketUDT#receive(ByteBuffer)
226         * @see com.barchart.udt.SocketUDT#receive(byte[], int, int)
227         */
228        @Override
229        public final int read(ByteBuffer buffer) throws IOException {
230
231                if (buffer == null) {
232                        throw new NullPointerException("buffer == null");
233                }
234
235                final int remaining = buffer.remaining();
236
237                if (remaining > 0) {
238
239                        final SocketUDT socket = socketUDT;
240                        final boolean isBlocking = isBlockingMode;
241
242                        final int sizeReceived;
243                        try {
244                                if (isBlocking) {
245                                        begin(); // JDK contract for NIO blocking calls
246                                }
247                                if (buffer.isDirect()) {
248                                        sizeReceived = socket.receive(buffer);
249                                } else {
250                                        assert buffer.hasArray();
251                                        byte[] array = buffer.array();
252                                        int position = buffer.position();
253                                        int limit = buffer.limit();
254                                        sizeReceived = socket.receive(array, position, limit);
255                                        if (0 < sizeReceived && sizeReceived <= remaining) {
256                                                buffer.position(position + sizeReceived);
257                                        }
258                                }
259                        } finally {
260                                if (isBlocking) {
261                                        end(true); // JDK contract for NIO blocking calls
262                                }
263                        }
264
265                        // see contract for receive()
266
267                        if (sizeReceived < 0) {
268                                log.trace("nothing was received; socketID={}",
269                                                socket.getSocketId());
270                                return 0;
271                        }
272
273                        if (sizeReceived == 0) {
274                                log.trace("receive timeout; socketID={}", socket.getSocketId());
275                                return 0;
276                        }
277
278                        if (sizeReceived <= remaining) {
279                                return sizeReceived;
280                        } else { // should not happen
281                                log.error("unexpected: sizeReceived > remaining; socketID={}",
282                                                socket.getSocketId());
283                                return 0;
284                        }
285
286                } else {
287                        return 0;
288                }
289        }

316        /**
317         * See {@link java.nio.channels.SocketChannel#write(ByteBuffer)} contract;
318         * 
319         * @return <code><0</code> should not happen<br>
320         *         <code>=0</code> blocking mode: timeout occurred on send<br>
321         *         <code>=0</code> non-blocking mode: buffer is full in the
322         *         underlying UDT socket; nothing is sent<br>
323         *         <code>>0</code> actual bytes sent count<br>
324         * @see com.barchart.udt.SocketUDT#send(ByteBuffer)
325         * @see com.barchart.udt.SocketUDT#send(byte[], int, int)
326         */
327        @Override
328        public final int write(ByteBuffer buffer) throws IOException {
329
330                // writeCount.incrementAndGet();
331
332                if (buffer == null) {
333                        throw new NullPointerException("buffer == null");
334                }
335
336                final int remaining = buffer.remaining();
337
338                if (remaining > 0) {
339
340                        final SocketUDT socket = socketUDT;
341                        final boolean isBlocking = isBlockingMode;
342
343                        final int sizeSent;
344                        try {
345                                if (isBlocking) {
346                                        begin(); // JDK contract for NIO blocking calls
347                                }
348                                if (buffer.isDirect()) {
349                                        sizeSent = socket.send(buffer);
350                                } else {
351                                        assert buffer.hasArray();
352                                        byte[] array = buffer.array();
353                                        int position = buffer.position();
354                                        int limit = buffer.limit();
355                                        sizeSent = socket.send(array, position, limit);
356                                        if (0 < sizeSent && sizeSent <= remaining) {
357                                                buffer.position(position + sizeSent);
358                                        }
359                                }
360                        } finally {
361                                if (isBlocking) {
362                                        end(true); // JDK contract for NIO blocking calls
363                                }
364                        }
365
366                        // see contract for send()
367
368                        if (sizeSent < 0) {
369                                log.trace("no buffer space for send; socketID={}",
370                                                socket.getSocketId());
371                                // logStatus();
372                                // log.info("writeCount={} writeSize={}", writeCount,
373                                // writeSize);
374                                // System.exit(1);
375                                return 0;
376                        }
377
378                        if (sizeSent == 0) {
379                                log.trace("send timeout; socketID={}", socket.getSocketId());
380                                return 0;
381                        }
382
383                        if (sizeSent <= remaining) {
384                                // writeSize.addAndGet(sizeSent);
385                                return sizeSent;
386                        } else { // should not happen
387                                log.error("unexpected: sizeSent > remaining; socketID={}",
388                                                socket.getSocketId());
389                                return 0;
390                        }
391
392                } else {
393                        return 0;
394                }
395        }
