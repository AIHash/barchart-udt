008    package com.barchart.udt.nio;
009    
010    import java.io.IOException;
011    import java.io.InputStream;
012    import java.io.OutputStream;
013    import java.nio.channels.SocketChannel;
014    
015    import com.barchart.udt.SocketUDT;
016    import com.barchart.udt.net.NetSocketUDT;
017    
018    public class AdapterSocketUDT extends NetSocketUDT {
019    
020            protected final ChannelSocketUDT channelUDT;
021    
022            protected AdapterSocketUDT( //
023                            final ChannelSocketUDT channelUDT, //
024                            final SocketUDT socketUDT //
025            ) {
026                    super(socketUDT);
027                    this.channelUDT = channelUDT;
028            }
029    
030            @Override
031            public ChannelSocketUDT getChannel() {
032                    return channelUDT;
033            }
034    
035            @Override
036            public synchronized InputStream getInputStream() throws IOException {
037                    if (inputStream == null) {
038                            inputStream = new AdapterInputStreamUDT(this.channelUDT, this);
039                    }
040                    return inputStream;
041            }
042    
043            @Override
044            public synchronized OutputStream getOutputStream() throws IOException {
045                    if (outputStream == null) {
046                            outputStream = new AdapterOutputStreamUDT(this.channelUDT, this);
047                    }
048                    return outputStream;
049            }
050    
051    }
