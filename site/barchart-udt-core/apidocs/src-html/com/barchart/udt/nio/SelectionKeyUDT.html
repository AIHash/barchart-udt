101            @Override
102            public synchronized SelectionKey interestOps(final int interestOps) {
103    
104                    assertValidKey();
105                    assertValidOps(interestOps);
106    
107                    try {
108    
109                            final Opt epollNew = from(interestOps);
110    
111                            // log.debug("epollNew : {}", epollNew);
112    
113                            if (epollNew != epollOpt) {
114    
115                                    // epollUDT().update(socketUDT(), epollNew);
116    
117                                    if (Opt.NONE == epollNew) {
118                                            epollUDT().remove(socketUDT());
119                                    } else {
120                                            epollUDT().remove(socketUDT());
121                                            epollUDT().add(socketUDT(), epollNew);
122                                    }
123    
124                                    epollOpt = epollNew;
125    
126                            }
127    
128                            // final Opt epollUdt = epollUDT().verify(socketUDT());
129                            // log.debug("epollUdt : {}", epollUdt);
130    
131                    } catch (final Exception e) {
132    
133                            log.error("epoll failure", e);
134    
135                    } finally {
136    
137                            this.interestOps = interestOps;
138    
139                    }
140    
141                    return this;
142    
143            }

174            /** read and write correlation counter */
175            private volatile int processCount;
176    
177            protected void logError(final String comment) {
178    
179                    final String message = "logic error : \n\t" + this;
180    
181                    log.warn(message, new Exception("" + comment));
182    
183            }

185            /**
186             * note: read interest is reported before write interest
187             * 
188             * @return should report state change?
189             */
190            protected boolean doRead(final int processCount) {
191    
192                    // log.debug("read  {}", this);
193    
194                    int readyOps = 0;
195                    final int interestOps = this.interestOps;
196                    this.processCount = processCount;
197    
198                    try {
199    
200                            if (!epollOpt.hasRead()) {
201                                    if (interestOps == 0) {
202                                            logError("error report when missing interest");
203                                            return false;
204                                    } else {
205                                            readyOps = interestOps;
206                                            return true;
207                                    }
208                            }
209    
210                            switch (kindUDT()) {
211                            case ACCEPTOR:
212                                    if (interestAccept(interestOps)) {
213                                            readyOps = OP_ACCEPT;
214                                            return true;
215                                    } else {
216                                            logError("ready to ACCEPT while not interested");
217                                            return false;
218                                    }
219                            case CONNECTOR:
220                                    if (interestRead(interestOps)) {
221                                            readyOps = OP_READ;
222                                            return true;
223                                    } else {
224                                            logError("ready to READ while not interested");
225                                            return false;
226                                    }
227                            default:
228                                    logError("wrong kind");
229                                    return false;
230                            }
231    
232                    } finally {
233    
234                            this.readyOps = readyOps;
235    
236                    }
237    
238            }
239    
240            /**
241             * note: write interest is reported after read interest
242             * 
243             * @return should report state change?
244             */
245            protected boolean doWrite(final int processCount) {
246    
247                    // log.debug("write {}", this);
248    
249                    int readyOps = 0;
250                    final int interestOps = this.interestOps;
251                    final boolean hadReadBeforeWrite = this.processCount == processCount;
252    
253                    try {
254    
255                            if (!epollOpt.hasWrite()) {
256                                    if (interestOps == 0) {
257                                            logError("error report when missing interest");
258                                            return false;
259                                    } else {
260                                            readyOps = interestOps;
261                                            return true;
262                                    }
263                            }
264    
265                            switch (kindUDT()) {
266                            case ACCEPTOR:
267                                    logError("ready to WRITE for acceptor");
268                                    return false;
269                            case CONNECTOR:
270                                    if (channelUDT().isConnectFinished()) {
271                                            if (interestWrite(interestOps)) {
272                                                    readyOps = OP_WRITE;
273                                                    return true;
274                                            } else {
275                                                    logError("ready to WRITE when not insterested");
276                                                    return false;
277                                            }
278                                    } else {
279                                            if (interestConnect(interestOps)) {
280                                                    readyOps = OP_CONNECT;
281                                                    return true;
282                                            } else {
283                                                    logError("ready to CONNECT when not interested");
284                                                    return false;
285                                            }
286                                    }
287                            default:
288                                    logError("wrong kind");
289                                    return false;
290                            }
291    
292                    } finally {
293                            if (hadReadBeforeWrite) {
294                                    this.readyOps |= readyOps;
295                            } else {
296                                    this.readyOps = readyOps;
297                            }
298                    }
299    
300            }

357            public static final String toStringOps(final int selectOps) {
358                    final char A = (OP_ACCEPT & selectOps) != 0 ? 'A' : '-';
359                    final char C = (OP_CONNECT & selectOps) != 0 ? 'C' : '-';
360                    final char R = (OP_READ & selectOps) != 0 ? 'R' : '-';
361                    final char W = (OP_WRITE & selectOps) != 0 ? 'W' : '-';
362                    return String.format("%c%c%c%c", A, C, R, W);
363            }
364    
365            protected static final int HAS_READ = OP_ACCEPT | OP_READ;
366            protected static final int HAS_WRITE = OP_CONNECT | OP_WRITE;
367    
368            /**
369             * select options : jdk to epoll
370             */
371            protected static Opt from(final int interestOps) {
372    
373                    final boolean hasRead = (interestOps & HAS_READ) != 0;
374                    final boolean hasWrite = (interestOps & HAS_WRITE) != 0;
375    
376                    if (hasRead && hasWrite) {
377                            return Opt.BOTH;
378                    }
379    
380                    if (hasRead) {
381                            return Opt.READ;
382                    }
383    
384                    if (hasWrite) {
385                            return Opt.WRITE;
386                    }
387    
388                    return Opt.NONE;
389    
390            }

392            /** is key not canceled? */
393            private volatile boolean isValid;
394    
395            @Override
396            public boolean isValid() {
397                    return isValid;
398            }
399    
400            /** add/remove poll/socket registration */
401            protected void makeValid(final boolean isValid) {
402                    try {
403                            if (isValid) {
404                                    epollOpt = Opt.NONE;
405                                    epollUDT().add(socketUDT(), epollOpt);
406                            } else {
407                                    epollUDT().remove(socketUDT());
408                            }
409                    } catch (final Exception e) {
410                            log.error("epoll failure", e);
411                    } finally {
412                            this.isValid = isValid;
413                    }
414            }
415    
416            @Override
417            public void cancel() {
418                    if (isValid()) {
419                            selectorUDT.cancel(this);
420                    }
421            }
422    
423            @Override
424            public int compareTo(final SelectionKeyUDT that) {
425                    final int thisId = this.socketId();
426                    final int thatId = that.socketId();
427                    if (thisId > thatId) {
428                            return +1;
429                    }
430                    if (thisId < thatId) {
431                            return -1;
432                    }
433                    return 0;
434            }
