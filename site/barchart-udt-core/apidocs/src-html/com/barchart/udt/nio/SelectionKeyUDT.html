008    package com.barchart.udt.nio;
009    
010    import java.net.InetSocketAddress;
011    import java.nio.channels.CancelledKeyException;
012    import java.nio.channels.SelectableChannel;
013    import java.nio.channels.SelectionKey;
014    import java.nio.channels.Selector;
015    import java.nio.channels.spi.AbstractSelectionKey;
016    
017    import com.barchart.udt.ExceptionUDT;
018    import com.barchart.udt.SocketUDT;
019    
020    /**
021     * NOTE: 1<->1 mapping between: SelectionKeyUDT -> ChannelUDT -> SocketUDT
022     */
023    public class SelectionKeyUDT extends AbstractSelectionKey {
024    
025            final SelectorUDT selectorUDT;
026            final ChannelUDT channelUDT;
027    
028            /**
029             * used by JNI
030             */
031            final int socketID;
032            final SocketUDT socketUDT;
033    
034            SelectionKeyUDT(SelectorUDT selector, ChannelUDT channel,
035                            Object attachment, int ops) {
036    
037                    this.channelUDT = channel;
038                    this.selectorUDT = selector;
039                    super.attach(attachment);
040                    this.interestOps = ops;
041    
042                    this.socketUDT = channel.getSocketUDT();
043                    this.socketID = socketUDT.getSocketId();
044    
045                    if (channel.getChannelKind() == KindUDT.CONNECTOR) {
046                            // special relationship for connectors
047                            ChannelSocketUDT socketChannel = (ChannelSocketUDT) channel;
048                            socketChannel.setRegistredKey(this);
049                    }
050    
051            }
052    
053            volatile int interestOps;
054            volatile int readyOps;
055    
056            //
057    
058            @Override
059            public SelectableChannel channel() {
060                    return (SelectableChannel) channelUDT;
061            }
062    
063            @Override
064            public int interestOps() {
065                    if (!isValid()) {
066                            throw new CancelledKeyException();
067                    }
068                    return interestOps;
069            }
070    
071            @Override
072            public SelectionKey interestOps(int ops) {
073                    if (!isValid()) {
074                            throw new CancelledKeyException();
075                    }
076                    if ((ops & ~(channel().validOps())) != 0) {
077                            throw new IllegalArgumentException("invalid ops");
078                    }
079                    if (((ops & OP_CONNECT) != 0) && socketUDT.isConnected()) {
080                            throw new IllegalArgumentException("already connected");
081                    }
082                    interestOps = ops;
083                    return this;
084            }
085    
086            @Override
087            public int readyOps() {
088                    if (!isValid()) {
089                            throw new CancelledKeyException();
090                    }
091                    return readyOps;
092            }
093    
094            @Override
095            public Selector selector() {
096                    return selectorUDT;
097            }
098    
099            //
100    
101            @Override
102            public int hashCode() {
103                    return socketID;
104            }
105    
106            @Override
107            public boolean equals(Object otherKey) {
108                    if (otherKey instanceof SelectionKeyUDT) {
109                            SelectionKeyUDT other = (SelectionKeyUDT) otherKey;
110                            return other.socketID == this.socketID;
111                    }
112                    return false;
113            }
114    
115            //
116    
117            @Override
118            public String toString() {
119                    InetSocketAddress local;
120                    try {
121                            local = socketUDT.getLocalSocketAddress();
122                    } catch (ExceptionUDT e) {
123                            local = null;
124                    }
125                    InetSocketAddress remote;
126                    try {
127                            remote = socketUDT.getRemoteSocketAddress();
128                    } catch (ExceptionUDT e) {
129                            remote = null;
130                    }
131    
132                    return " socketID=" + socketID
133                                    + //
134                                    " type=" + channelUDT.getChannelKind()
135                                    + //
136                                    " readyOps=" + toStringOps(readyOps) + "(" + readyOps + ")"
137                                    + //
138                                    " ntrstOps=" + toStringOps(interestOps) + "(" + interestOps
139                                    + ")" + //
140                                    " local=" + local + //
141                                    " remote=" + remote + //
142                                    "";
143            }
144    
145            public static final String toStringOps(int keyOps) {
146                    char R = (OP_READ & keyOps) != 0 ? 'R' : '-';
147                    char W = (OP_WRITE & keyOps) != 0 ? 'W' : '-';
148                    char C = (OP_CONNECT & keyOps) != 0 ? 'C' : '-';
149                    char A = (OP_ACCEPT & keyOps) != 0 ? 'A' : '-';
150                    return String.format("%c%c%c%c", A, C, W, R);
151            }
152    
153    }
