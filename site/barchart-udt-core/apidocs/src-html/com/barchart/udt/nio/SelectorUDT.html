082        protected static Selector open(TypeUDT type) throws IOException {
083                final SelectorProviderUDT provider;
084                switch (type) {
085                case DATAGRAM:
086                        provider = SelectorProviderUDT.DATAGRAM;
087                        break;
088                case STREAM:
089                        provider = SelectorProviderUDT.STREAM;
090                        break;
091                default:
092                        log.error("unsupported type={}", type);
093                        throw new IOException("unsupported type");
094                }
095                return provider.openSelector();
096        }

100        @Override
101        protected void implCloseSelector() throws IOException {
102                wakeup();
103                synchronized (this) {
104                        synchronized (publicRegisteredKeySet) {
105                                synchronized (publicSelectedKeySet) {
106
107                                        selectedKeySet.clear();
108
109                                        for (final SelectionKey key : registeredKeySet) {
110                                                try {
111                                                        key.channel().close();
112                                                } catch (Throwable e) {
113                                                        log.error("unexpected", e);
114                                                }
115                                        }
116
117                                        registeredKeySet.clear();
118
119                                }
120                        }
121                }
122        }

124        // NOE: register() and select() are blocking each other
125        @Override
126        protected SelectionKey register(AbstractSelectableChannel channel,
127                        int interestOps, Object attachment) {
128
129                if (!(channel instanceof ChannelUDT)) {
130                        // also takes care of null
131                        log.error("!(channel instanceof ChannelUDT)");
132                        throw new IllegalSelectorException();
133                }
134
135                synchronized (publicRegisteredKeySet) {
136
137                        if (publicRegisteredKeySet.size() == maximimSelectorSize) {
138                                log.error("reached maximimSelectorSize)");
139                                throw new IllegalSelectorException();
140                        }
141
142                        ChannelUDT channelUDT = (ChannelUDT) channel;
143
144                        SocketUDT socketUDT = channelUDT.getSocketUDT();
145
146                        SelectionKeyUDT keyUDT = new SelectionKeyUDT(//
147                                        this, channelUDT, attachment, interestOps);
148
149                        // XXX the only place with "add/put"
150                        registeredKeyMap.put(socketUDT.getSocketId(), keyUDT);
151                        registeredKeySet.add(keyUDT);
152
153                        return keyUDT;
154
155                }
156
157        }

182        @Override
183        // per java.nio.Selector contract 0 input means infinite
184        public int select(long timeout) throws IOException {
185                if (timeout < 0) {
186                        throw new IllegalArgumentException("Negative timeout");
187                }
188                long timeoutUDT = (timeout == 0) ? UDT_TIMEOUT_INFINITE : timeout;
189                return doSelectInsideLock(timeoutUDT);
190        }
191
192        @Override
193        public int selectNow() throws IOException {
194                return doSelectInsideLock(UDT_TIMEOUT_NONE);
195        }

197        /**
198         * Wakeup the selector. NOTE: it is recommended to user timed
199         * {@link #select(long)} (say, timeout = 100 milliseconds) to avoid missed
200         * {@link #wakeup()} calls which are rare but possible in current
201         * implementation (accepted in favor of design simplicity / performance)
202         */
203        @Override
204        public Selector wakeup() {
205                // publisher for volatile
206                wakeupStepCount++;
207                return this;
208        }

210        private volatile int wakeupStepCount;
211
212        // guarded by doSelectInsideLock
213        private int wakeupBaseCount;
214
215        private final void saveWakeupBase() {
216                wakeupBaseCount = wakeupStepCount;
217        }
218
219        private final boolean isWakeupPending() {
220                return wakeupBaseCount != wakeupStepCount;
221        }

270        SelectorUDT(SelectorProvider provider, //
271                        int maximumSelectorSize, int maximumConnectorSize) {
272
273                super(provider);
274
275                registeredKeyMap = new HashMap<Integer, SelectionKeyUDT>();
276
277                registeredKeySet = new HashSet<SelectionKeyUDT>();
278                selectedKeySet = new HashSet<SelectionKeyUDT>();
279                cancelledKeySet = cancelledKeys();
280
281                publicRegisteredKeySet = Collections.unmodifiableSet(registeredKeySet);
282                publicSelectedKeySet = HelperNIOUDT.ungrowableSet(selectedKeySet);
283
284                this.maximimSelectorSize = maximumSelectorSize;
285                if (isBufferBased) {
286                        readBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
287                        writeBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
288                        exceptBuffer = SocketUDT.newDirectIntBufer(maximumSelectorSize);
289                        sizeBuffer = SocketUDT.newDirectIntBufer(UDT_SIZE_COUNT);
290                        readArray = null;
291                        writeArray = null;
292                        exceptArray = null;
293                        sizeArray = null;
294                } else {
295                        readBuffer = null;
296                        writeBuffer = null;
297                        exceptBuffer = null;
298                        sizeBuffer = null;
299                        readArray = new int[maximumSelectorSize];
300                        writeArray = new int[maximumSelectorSize];
301                        exceptArray = new int[maximumSelectorSize];
302                        sizeArray = new int[UDT_SIZE_COUNT];
303                }
304
305                this.maximumConnectorSize = maximumConnectorSize;
306                connectorPool = new ConnectorThreadPoolUDT(maximumConnectorSize);
307

314        private final int doSelectInsideLock(long millisTimeout) throws IOException {
315                if (!isOpen()) {
316                        throw new ClosedSelectorException();
317                }
318                synchronized (this) {
319                        synchronized (publicRegisteredKeySet) {
320                                synchronized (publicSelectedKeySet) {
321                                        return doSelectReally(millisTimeout);
322                                }
323                        }
324                }
325        }

393                if (isBufferBased) {
394                        return SocketUDT.select(readBuffer, writeBuffer, exceptBuffer,

439        private final void prepareInterest() {
440
441                int readSize = 0;
442                int writeSize = 0;
443                int exceptSize = 0;
444
445                for (final SelectionKeyUDT keyUDT : registeredKeySet) {
446
447                        final ChannelUDT channelUDT = keyUDT.channelUDT;
448                        keyUDT.readyOps = 0; // publisher for volatile
449
450                        if (channelUDT.isOpenSocketUDT()) {
451
452                                final int interestOps = keyUDT.interestOps;
453                                final int socketID = keyUDT.socketID;
454                                final KindUDT channelType = channelUDT.getChannelKind();
455
456                                if ((interestOps & (OP_ACCEPT)) != 0) {
457                                        assert channelType == KindUDT.ACCEPTOR;
458                                        setReadInterest(readSize, socketID);
459                                        readSize++;
460                                }
461                                if ((interestOps & (OP_READ)) != 0) {
462                                        assert channelType == KindUDT.CONNECTOR;
463                                        setReadInterest(readSize, socketID);
464                                        readSize++;
465                                }
466                                if ((interestOps & (OP_WRITE)) != 0) {
467                                        assert channelType == KindUDT.CONNECTOR;
468                                        setWriteInterest(writeSize, socketID);
469                                        writeSize++;
470                                }
471                                if ((interestOps & (OP_CONNECT)) != 0) {
472                                        assert channelType == KindUDT.CONNECTOR;
473                                        assert (interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
474                                        /*
475                                         * UDT does not support select() for connect() operation as
476                                         * yet; using thread pool instead
477                                         */
478                                }
479                        } else {
480                                synchronized (cancelledKeySet) {
481                                        cancelledKeySet.add(keyUDT);
482                                }
483                        }
484                }
485
486                // set sizes
487                setInterestSize(readSize, writeSize, exceptSize);
488
489        }
490
491        private final void processInterest() {
492
493                // add ready to selected
494                updateRead();
495                updateWrite();
496                updateExcept();
497
498        }

500        /**
501         * note: millisTimeout input values contract:
502         * 
503         * -1 : infinite
504         * 
505         * =0 : immediate (but really up to 10 ms UDT resolution slice)
506         * 
507         * >0 : finite
508         */
509        private final int performSelect(long millisTimeout) throws ExceptionUDT {
510                int readyCount = 0;
511                saveWakeupBase();
512                if (millisTimeout < 0) {
513                        /* infinite: do select in slices; check for wakeup; */
514                        do {
515                                readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
516                                if (readyCount > 0 || isWakeupPending()) {
517                                        break;
518                                }
519                        } while (true);
520                } else if (millisTimeout > 0) {
521                        /* finite: do select in slices; check for wakeup; count down */
522                        do {
523                                readyCount = doSelectSocketUDT(DEFAULT_MIN_SELECTOR_TIMEOUT);
524                                if (readyCount > 0 || isWakeupPending()) {
525                                        break;
526                                }
527                                millisTimeout -= DEFAULT_MIN_SELECTOR_TIMEOUT;
528                        } while (millisTimeout > 0);
529                } else { // millisTimeout == 0
530                        /* immediate: one shot select */
531                        readyCount = doSelectSocketUDT(0);
532                }
533                return readyCount;
534        }
535
536        /**
537         * note: return value contract:
538         * 
539         * <0 : should not happen
540         * 
541         * =0 : means nothing was selected/timeout
542         * 
543         * >0 : number of pending r/w ops, NOT number of selected keys
544         */
545        private final int doSelectReally(long millisTimeout) throws IOException {
546
547                selectedKeySet.clear();
548
549                processCancelled();
550
551                int readyCount = 0;
552
553                trySelect: try {
554                        // java.nio.Selector contract for wakeup()
555                        begin();
556                        // pre select
557                        prepareInterest();
558                        // into select
559                        readyCount = performSelect(millisTimeout);
560                        if (readyCount == 0) {
561                                // timeout, nothing is ready; no need to process post select
562                                break trySelect;
563                        }
564                        // post select
565                        processInterest();
566                } finally {
567                        // java.nio.Selector contract for wakeup()
568                        end();
569                }
570
571                // using thread pool based connect() processor
572                readyCount += updateConnect();
573
574                assert readyCount >= 0;
575
576                return readyCount;
577
578        }

580        private final int updateConnect() {
581                final Queue<SelectionKeyUDT> readyQueue = connectorPool.readyQueue;
582                if (readyQueue.isEmpty()) {
583                        return 0;
584                }
585                int updateCount = 0;
586                SelectionKeyUDT keyUDT;
587                while ((keyUDT = readyQueue.poll()) != null) {
588                        // contract:
589                        assert keyUDT.channelUDT.getChannelKind() == KindUDT.CONNECTOR;
590                        assert registeredKeySet.contains(keyUDT);
591                        assert (keyUDT.interestOps & OP_CONNECT) != 0;
592                        assert (keyUDT.interestOps & (OP_READ | OP_WRITE | OP_ACCEPT)) == 0;
593                        //
594                        keyUDT.readyOps |= OP_CONNECT;
595                        selectedKeySet.add(keyUDT);
596                        updateCount++;
597                }
598                return updateCount;
599        }

601        private final void updateRead() {
602                final int readSize = getReadInterestSize();
603                for (int index = 0; index < readSize; index++) {
604                        final int socketId = getReadInterest(index);
605                        final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
606                        assert keyUDT != null;
607                        switch (keyUDT.channelUDT.getChannelKind()) {
608                        case ACCEPTOR:
609                                keyUDT.readyOps |= OP_ACCEPT;
610                                break;
611                        case CONNECTOR:
612                                keyUDT.readyOps |= OP_READ;
613                                break;
614                        default:
615                                assert false : "unexpected default";
616                                continue;
617                        }
618                        selectedKeySet.add(keyUDT);
619                }
620        }
621
622        private final void updateWrite() {
623                final int writeSize = getWriteInterestSize();
624                for (int index = 0; index < writeSize; index++) {
625                        final int socketId = getWriteInterest(index);
626                        final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
627                        assert keyUDT != null;
628                        switch (keyUDT.channelUDT.getChannelKind()) {
629                        case ACCEPTOR:
630                                assert false : "unexpected ACCEPTOR";
631                                continue;
632                        case CONNECTOR:
633                                keyUDT.readyOps |= OP_WRITE;
634                                break;
635                        default:
636                                assert false : "unexpected default";
637                                continue;
638                        }
639                        selectedKeySet.add(keyUDT);
640                }
641        }
642
643        private final void updateExcept() {
644                final int exceptSize = getExceptInterestSize();
645                for (int k = 0; k < exceptSize; k++) {
646                        final int socketId = getExceptInterest(k);
647                        final SelectionKeyUDT keyUDT = registeredKeyMap.get(socketId);
648                        assert keyUDT != null;
649                        switch (keyUDT.channelUDT.getChannelKind()) {
650                        case ACCEPTOR:
651                        case CONNECTOR:
652                                // set all ready OPS to throw on any operation
653                                keyUDT.readyOps |= keyUDT.channel().validOps();
654                                break;
655                        default:
656                                assert false : "unexpected default";
657                                continue;
658                        }
659                        selectedKeySet.add(keyUDT);
660                }
661        }
662
663        private final void processCancelled() throws IOException {
664                /*
665                 * Precondition: Synchronized on this, publicRegisteredKeySet,
666                 * publicSelectedKeySet
667                 */
668                synchronized (cancelledKeySet) {
669                        if (!cancelledKeySet.isEmpty()) {
670                                for (final SelectionKey key : cancelledKeySet) {
671
672                                        final SelectionKeyUDT keyUDT = (SelectionKeyUDT) key;
673
674                                        // XXX the only place with "remove"
675
676                                        final SelectionKeyUDT removed = //
677                                        registeredKeyMap.remove(keyUDT.socketID);
678                                        assert removed != null;
679
680                                        final boolean isRemoved = //
681                                        registeredKeySet.remove(keyUDT);
682                                        assert isRemoved;
683
684                                }
685                                cancelledKeySet.clear();
686                        }
687                }
688        }
689
690        final void submitConnectRequest(SelectionKeyUDT keyUDT,
691                        InetSocketAddress remote) throws IOException {
692
693                // TODO think again if lack of sync is OK?
694
695                if (!registeredKeySet.contains(keyUDT)) {
696                        throw new IOException(//
697                                        "connect request while not registered");
698                }
699
700                if ((keyUDT.interestOps & OP_CONNECT) == 0) {
701                        throw new IOException("connect request while not interested;"
702                                        + " key=" + keyUDT);
703                }
704
705                if ((keyUDT.interestOps & (OP_ACCEPT | OP_READ | OP_WRITE)) != 0) {
706                        throw new IOException("connect request while is not sole interest;"
707                                        + " key=" + keyUDT);
708                }
709
710                connectorPool.submitRequest(keyUDT, remote);
711
712        }
