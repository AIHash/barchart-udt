18  /**
19   * selection provider for UDT
20   * <p>
21   * note: you must use the same system-wide provider instance for the same
22   * {@link TypeUDT} of UDT channels and UDT selectors;
23   */
24  public class SelectorProviderUDT extends SelectorProvider {
25  
26  	/**
27  	 * system-wide provider instance, for {@link TypeUDT#DATAGRAM} UDT sockets
28  	 */
29  	public static final SelectorProviderUDT DATAGRAM = //
30  	new SelectorProviderUDT(TypeUDT.DATAGRAM);
31  
32  	/**
33  	 * system-wide provider instance, for {@link TypeUDT#STREAM} UDT sockets
34  	 */
35  	public static final SelectorProviderUDT STREAM = //
36  	new SelectorProviderUDT(TypeUDT.STREAM);
37  
38  	public static SelectorProviderUDT from(final TypeUDT type) {
39  		switch (type) {
40  		case DATAGRAM:
41  			return DATAGRAM;
42  		case STREAM:
43  			return STREAM;
44  		default:
45  			throw new IllegalStateException("wrong type=" + type);
46  		}
47  	}
48

51  	private volatile int maxSelectorSize = SocketUDT.DEFAULT_MAX_SELECTOR_SIZE;
52  
53  	private final TypeUDT type;
54  
55  	/**
56  	 * {@link TypeUDT} of UDT sockets generated by this provider
57  	 */
58  	public final TypeUDT type() {
59  		return type;
60  	}
61  
62  	protected SelectorProviderUDT(final TypeUDT type) {

64  	}
