159 	public static void appendSnapshot(final SocketUDT socketUDT,
160 			final StringBuilder text) {
161 
162 		text.append("\n\t");
163 		text.append("socketID");
164 		text.append(" = ");
165 		text.append(socketUDT.socketID);
166 
167 		for (OptionUDT option : values()) {
168 			int optionCode = 0;
169 			String optionName = null;
170 			String optionValue = null;
171 			try {
172 
173 				optionCode = option.code;
174 				optionName = option.name();
175 				optionValue = option.format.convert(//
176 						socketUDT.getOption(option));
177 
178 				// TODO fix this hack
179 				if (optionName.startsWith("UD")) {
180 					continue;
181 				}
182 
183 				text.append("\n\t");
184 				text.append(optionCode);
185 				text.append(") ");
186 				text.append(optionName);
187 				text.append(" = ");
188 				text.append(optionValue);
189 
190 			} catch (Exception e) {
191 				log.error("unexpected; " + optionName, e);
192 			}
193 
194 		}
