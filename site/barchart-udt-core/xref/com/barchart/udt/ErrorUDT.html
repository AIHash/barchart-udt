114 	static final ErrorUDT[] ENUM_VALS = values();
115 
116 	public static ErrorUDT errorFrom(final int code) {
117 		for (final ErrorUDT known : ENUM_VALS) {
118 			if (known.code == code) {
119 				return known;
120 			}
121 		}
122 		return WRAPPER_UNKNOWN;
123 	}
