008    package com.barchart.udt.net;
009    
010    import java.io.IOException;
011    import java.net.InetAddress;
012    import java.net.InetSocketAddress;
013    import java.net.ServerSocket;
014    import java.net.Socket;
015    import java.net.SocketAddress;
016    import java.net.SocketException;
017    import java.nio.channels.ServerSocketChannel;
018    
019    import com.barchart.udt.ExceptionUDT;
020    import com.barchart.udt.SocketUDT;
021    import com.barchart.udt.TypeUDT;
022    
023    public class NetServerSocketUDT extends ServerSocket implements IceServerSocket {
024    
025            protected final SocketUDT serverSocketUDT;
026    
027            public NetServerSocketUDT() throws IOException {
028                    this.serverSocketUDT = new SocketUDT(TypeUDT.STREAM);
029                    this.serverSocketUDT.configureBlocking(true);
030            }
031    
032            // exception thanks to JDK designers
033            /** NOTE: you just carefully choose TypeUDT */
034            public NetServerSocketUDT(final SocketUDT socketUDT) throws IOException {
035                    this.serverSocketUDT = socketUDT;
036            }
037    
038            //
039    
040            @Override
041            public Socket accept() throws IOException {
042                    final SocketUDT connector = serverSocketUDT.accept();
043                    return new NetSocketUDT(connector);
044            }
045    
046            @Override
047            public void bind(SocketAddress endpoint) throws IOException {
048                    final int backlog = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
049                    bind(endpoint, backlog);
050            }
051    
052            // NOTE: bind() means listen() for UDT server socket
053            /*
054             * The listen method lets a UDT socket enter listening state. The socket
055             * must call bind before a listen call.
056             */
057            @Override
058            public void bind(SocketAddress bindpoint, int backlog) throws IOException {
059                    if (bindpoint == null) {
060                            bindpoint = new InetSocketAddress(0);
061                    }
062                    if (backlog <= 0) {
063                            backlog = SocketUDT.DEFAULT_ACCEPT_QUEUE_SIZE;
064                    }
065                    serverSocketUDT.bind((InetSocketAddress) bindpoint);
066                    serverSocketUDT.listen(backlog);
067            }
068    
069            @Override
070            public void close() throws IOException {
071                    serverSocketUDT.close();
072            }
073    
074            @Override
075            public ServerSocketChannel getChannel() {
076                    throw new RuntimeException("feature not available");
077            }
078    
079            @Override
080            public InetAddress getInetAddress() {
081                    return serverSocketUDT.getLocalInetAddress();
082            }
083    
084            @Override
085            public int getLocalPort() {
086                    return serverSocketUDT.getLocalInetPort();
087            }
088    
089            @Override
090            public SocketAddress getLocalSocketAddress() {
091                    try {
092                            return serverSocketUDT.getLocalSocketAddress();
093                    } catch (ExceptionUDT e) {
094                            return null;
095                    }
096            }
097    
098            @Override
099            public int getReceiveBufferSize() throws SocketException {
100                    return serverSocketUDT.getReceiveBufferSize();
101            }
102    
103            @Override
104            public boolean getReuseAddress() throws SocketException {
105                    return serverSocketUDT.getReuseAddress();
106            }
107    
108            @Override
109            public int getSoTimeout() throws IOException {
110                    return serverSocketUDT.getSoTimeout();
111            }
112    
113            @Override
114            public boolean isBound() {
115                    return serverSocketUDT.isBound();
116            }
117    
118            @Override
119            public boolean isClosed() {
120                    return serverSocketUDT.isClosed();
121            }
122    
123            @Override
124            public void setPerformancePreferences(int connectionTime, int latency,
125                            int bandwidth) {
126                    throw new RuntimeException("feature not available");
127            }
128    
129            // NOTE: set both send and receive, since they are inherited on accept()
130            @Override
131            public void setReceiveBufferSize(int size) throws SocketException {
132                    serverSocketUDT.setReceiveBufferSize(size);
133                    serverSocketUDT.setSendBufferSize(size);
134            }
135    
136            @Override
137            public void setReuseAddress(boolean on) throws SocketException {
138                    serverSocketUDT.setReuseAddress(on);
139            }
140    
141            @Override
142            public void setSoTimeout(int timeout) throws SocketException {
143                    serverSocketUDT.setSoTimeout(timeout);
144            }
145    
146    }
